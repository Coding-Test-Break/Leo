N = int(input())

def qualifier(answer):
    open_cnt = 0
    for i in answer:
        if i == "(":
            open_cnt += 1
        elif i == ")":
            open_cnt -= 1

        if open_cnt < 0:
            return "NO"
    else: 
        if open_cnt == 0:
            return "YES"
        else: return "NO"

for i in range(N):
    answer = input()
    print(qualifier(answer))
